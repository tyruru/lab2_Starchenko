def ispalindrom(text):
    text = text.replace(" ", "").lower()
    if text == text[::-1]:
        print("It's a palindrom")
    else:
        print("It's not a palindrom")

def FindWord(word, string):
    index = 0
    for letter in word:
        index = string.find(letter, index)
        if (index == -1):
            return False
        index += 1
    
    return True
